第一部分：数据库创建

1.1:数据库的创建与维护

1.1创建一个名为StudentDB的数据库，该数据库包含相应的主数据文件、日志文件、和名为stuDB1和stuDB2的辅助数据文件；主数据文件和日志文件建立在：”D:\DATA“目录下，辅助数据文件建立在：”E:\DATA”目录下；数据文件初始大小为：3M,按1M无限制增长；日志文件按%10有限制增长，文件最大空间为1GB。

create database StudentDB
on primary
(
    name='StudentDB',
    filename='D:\DATA\StudentDB.mdf',
    size=3mb,
    maxsize=unlimited,
    filegrowth=1mb
),
(
    name='StuDB1',
    filename='E:\DATA\StuDB1.ndf',
    size=3mb,
    maxsize=unlimited,
    filegrowth=1mb
),
(
    name='StuDB2',
    filename='E:\DATA\StuDB2.ndf',
    size=3mb,
    maxsize=unlimited,
    filegrowth=1mb
)
log on
(
    name='StudentDB_log',
    filename='D:\DATA\StudentDB_log.ldf',
    size=1mb,
    maxsize=1gb,
    filegrowth=10%
)
1.2:修改数据库

1.修改StudentDB数据库主数据文件的初始大小为5M;

2.在StudentDB数据库中创建名为”GData”的文件组，为数据库添加辅助数据文件”StuDB3”,文件建立在”E:DATA”目录下，将该文件添加到”GData”文件组中。

1.
alter database StudentDB

name=StudentDB,

2.1：添加、修改与删除数据表中的某列
alter table [表名]
add [字段名] [字段属性]

alter table [表名]
alter column [字段名] [字段新属性]

alter table [表名]

2.2：创建分区表

在StuDB中，依据表Student表的StudentYear列，来创建分区函数YearOrderPartFunc,

边界值为2007和2009

--首先创建分区函数
create partition function YearOrderPartFunc([字段类型/int/varchar])
as range left for values('2007','2009')

--然后创建分区方案
create partition scheme YearOrderPartScheme
as partition YearOrderPartFunc
to ([primary],[文件组名],[文件组名])

--最后创建表
create table [表名]
(字段……略
StdYear in not null
)
on YearOrderPartScheme(StdYear)

--查看数据所在分区
select *,$partition.YearOrderPartFunc(StdYear)
as '所属分区'
from [表名]
2.3：表的约束
-- 修改表的时候定义约束
-- 给某个字段增加默认值default
alter table [表名]
add constraint [约束名] default([约束条件]) for [字段名]

-- 字段值取值唯一
alter table [表名]
add constraint [约束名] unique([字段名])

-- 检查约束
alter table [表名]
add constraint [约束名] check([字段名]>=12 and [字段名]<67)

-- 主键约束
alter table [表名]
add constraint [约束名] primary key([字段名])

-- 外键约束
[字段名] [字段类型] foreign key references [表名]([外键字段名])  // 建表时写法
alter table [表名]
add constraint [约束名] foreign key([字段名]) references [表名]([外键字段名])

-- 删除约束
alter table [表名]
drop constraint [约束名]
2.4：索引
-- 聚集索引
create clustered index [索引名]
on [表名]([字段名] desc/asc)

-- 重新生成索引
alter index [索引名]
on [表名] rebuild

-- 删除索引
drop index [索引名]
on [表名]
2.5：查询
-- 消除重复查询
select distinct [要查询的字段名，不同字段名中间用逗号隔开] from [表名]

-- 查找前n条记录
select top n [字段名] from [表名]

-- 查找前n%记录
select top n percent [字段名] from [表名]

-- 赋予记录中文别名
select  [字段名] as [自定义属性名]
from [表名]

-- ‘+’号，若两个字段都为字符串类型作拼接操作，若为数字型字符串和int型字段则进行相加，都为int型进行相加；
select [字段名]+[字段名] from [表名]

-- 查询结果按条件排序
select [字段名] from [表名]
where [查询条件]
order by [字段名] asc  -- 不写默认为升序asc ，降序需写明为desc，可依据多个字段排序，如：order by [字段名1] asc, [字段名2] desc

-- 聚合函数
-- 统计该表记录总数
select COUNT(*) from [表名]

-- 统计总数、最大、最小以及平均值
select SUM([字段名]) as 总数,
       MAX([字段名]) as 最大值,
       MIN([字段名]) as 最小值,
       AVG([字段名]) as 平均值
from [表名]

-- 按某个字段查询该字段不同记录个数
select [字段名A]，COUNT([字段名A]) as 数量
from [表名]
group by [字段名A]

-- 显示信息，并汇总指定字段总数
select [字段名1，字段名2] from [表名]
compute count(字段名1)

-- 显示信息，并按指定字段的分类汇总
select [字段名1，字段名2] from [表名]
order by [字段名2]
compute count([字段名1]) by [字段名2]

-- 内连接查询(返回所有满足条件的记录)
select [字段名1],[字段名2],[表名1].[共同字段名] 
from [表名1] inner join [表名2]
on [表名1].[共同字段名]=[表名2].[共同字段名]
where [条件]
-- 第二种方式（别名方式）
select [字段名1],[字段名2],[表名1].[共同字段名] 
from [表名1] a  join [表名2] b
on a.[共同字段名]=b.[共同字段名]
where [条件]

-- 左/右外连接（包括左/右表不满足条件的记录，也就是说不管条件如何都显示左/右表所有记录）
select [字段名1],[字段名2],[表名1].[共同字段名] 
from [表名1] a left/right join [表名2] b
on a.[共同字段名]=b.[共同字段名]
where [条件]

-- 完全外连接（不管条件如何，包括左表和右表所有的记录行）
select [字段名1],[字段名2],[表名1].[共同字段名] 
from [表名1] a full join [表名2] b
on a.[共同字段名]=b.[共同字段名]
where [条件]

-- 三表连接
select [字段名1],[字段名2]
from [表名1] a full join [表名2] b
on a.[共同字段名]=b.[共同字段名] join [表名3] c
on b.[共同字段名]=c.[共同字段名]
where [条件]

-- 并运算查询，把两个查询的结果合并在一起，结果只会显示前面字段名
select [字段名1] 别名 from [表名1]
union
select [字段名2] from [表名2]

-- 交运算，输出两个查询结果的交集
select [字段名1] 别名 from [表名1]
intersect
select [字段名2] from [表名2]

-- 差运算，输出两个查询结果的差集
select [字段名1] 别名 from [表名1]
except
select [字段名2] from [表名2]

第三部分：SQL编程

3.1：存储过程

3.1.1 查询StdInfoID为6的学生的姓名和已选课程门数，当选课门数在3门以上，输出”XX，已经完成了选课”，否则输出”XX，还需选课”。

declare @sname varchar(50),@num int /*声明变量*/
select @sname=StdInfoName from StudentInfo
where StdInfoID=6
set @num=(select COUNT(*)) from [StudentInfo]
    group  by StdInfoID
    having StdInfoID=6)
if @num>=3
    begin 
        print @sname+',已经完成选课！'
    end
else
    print @sname+',选课门数不够！'

3.1.2 简单CASE结构：查询StudentInfo表中学生的性别，若是女生显示”female”，否则显示”male”。

select StdInfoName, StdInfoSex, 性别=case
        when StdInfoSex='男' then 'male'
        when StdInfoSex='女' then 'female'
        end
    from StudentInfo

3.1.3 循环结构，

计算1-100内能被3整除的最大数。

declare @x int=100
while(@x>=1)
    begin
        if(@x%3=0)
            break
        set@x-=1
    end
print @x

3.2：DML触发器

3.2.1：创建INSERT触发器，如果向教师信息表TeachInfo中插入记录时，检查该记录的院系编号在院系信息表DepInfo中是否存在，如果不存在，则不允许插入

-- inserted 逻辑表， 存放插入之后的记录
create trigger trig_insertTeachinfo
on TeachInfo
after insert
as
begin insert
if exists (select * from inserted  a
          where a.DepInfoID not in(
          select DepInfoID from DepInfo))
begin 
    raiserror('数据一致性验证！',16,1)         /* 报错 */
    rollback transaction                 /* 事务回滚 */
end
else
    commit transaction

3.2.2.创建UPDATE触发器，当班级信息表ClassInfo中的主键ClassInfoID进行修改时，必须对学生表中相应的ClassInfoID也进行修改

create trigger trig_updateClass
on ClassInfo
after update
if update(ClassInfoID)
begin 
    update StudentInfo
    set ClassInfoID=(select ClassInfoID from inserted)
    where ClassInfoID=(select ClassInfoID from deleted)
end

3.2.3. 创建DELETE触发器，当删除学生信息表StudentInfo中已退学的学生资料时，自动删除学生选课表

create trigger trig_deleteStudent
on StudentInfo
after delete
as
begin
    declare @stdInfoID int
    select @stdInfoID=deleted.StdInfoID from deleted
    delete StudentCourse
    where StdInfoID = @stdInfoID
end
